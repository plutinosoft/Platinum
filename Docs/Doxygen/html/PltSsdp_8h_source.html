261 class PLT_NetworkInterfaceAddressSearchIterator
262 {
263 public:
264  PLT_NetworkInterfaceAddressSearchIterator(NPT_String ip) : m_Ip(ip) {}
265  virtual ~PLT_NetworkInterfaceAddressSearchIterator() {}
266 
267  NPT_Result operator()(NPT_NetworkInterface*& addr) const {
268  NPT_List<NPT_NetworkInterfaceAddress>::Iterator niaddr = addr->GetAddresses().GetFirstItem();
269  if (!niaddr) return NPT_FAILURE;
270 
271  return (m_Ip.Compare((*niaddr).GetPrimaryAddress().ToString(), true) == 0) ? NPT_SUCCESS : NPT_FAILURE;
272  }
273 
274 private:
275  NPT_String m_Ip;
276 };
