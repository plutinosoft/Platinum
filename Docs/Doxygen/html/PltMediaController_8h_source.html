313  static void ParseCSV(const char* csv, PLT_StringList& values) {
314  const char* start = csv;
315  const char* p = start;
316 
317  // look for the , character
318  while (*p) {
319  if (*p == ',') {
320  NPT_String val(start, (int)(p-start));
321  val.Trim(' ');
322  values.Add(val);
323  start = p + 1;
324  }
325  p++;
326  }
327 
328  // last one
329  NPT_String last(start, (int)(p-start));
330  last.Trim(' ');
331  if (last.GetLength()) {
332  values.Add(last);
333  }
334  }
